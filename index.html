Vue 的使用入門

  • 安裝 Vuex：
    npm install vuex
  • 創建一個 Vuex Store：
    store/index.js
    
import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export default new Vuex.Store({
  state: {
    count: 0,
  },
  mutations: {
    increment(state) {
      state.count++;
    },
  },
  actions: {
    increment({ commit }) {
      commit('increment');
    },
  },
  getters: {
    doubleCount(state) {
      return state.count * 2;
    },
  },
});
  • 在主 Vue 實例中使用 Vuex Store：
    main.js
    import Vue from 'vue';
import App from './App.vue';
import store from './store';

new Vue({
  store,
  render: h => h(App),
}).$mount('#app');
  • 從 Store 中獲取狀態：
    computed: {
  count() {
    return this.$store.state.count;
  },
},
    提交 mutation：
    methods: {
  increment() {
    this.$store.commit('increment');
  },
},
    調用 action：
    methods: {
  increment() {
    this.$store.dispatch('increment');
  },
},
    使用 getters：
    computed: {
  doubleCount() {
    return this.$store.getters.doubleCount;
  },
},
